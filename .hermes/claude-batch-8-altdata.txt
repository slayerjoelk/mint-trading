=== FILE: core/alternative_data.py ===

An alternative data pipeline that supplements price data with additional signals.

class AlternativeData:
    def __init__(self, data_pipeline, db):
        ...
    
    def get_options_flow(self, ticker: str) -> dict:
        """Proxy for put/call sentiment. Use yfinance to pull option chain data.
        Returns: put_call_ratio, open_interest, unusual_activity_score"""
    
    def get_sector_rotation(self) -> dict:
        """Compare sector ETF performance: XLK, XLF, XLE, XLV, XLI, XLY, XLRE, XLU, XLB, XLP.
        Returns: top_sectors (best 3 performers in last 20 days),
        bottom_sectors (worst 3), rotation_signal (where money is flowing)"""
    
    def get_market_breadth(self) -> dict:
        """IWM vs SPY, advance/decline proxy via RSP vs SPY trend.
        Returns: breadth_health, participation_score, divergence_warning"""
    
    def get_economic_surprise_proxy(self) -> float:
        """Proxy for economic surprise index using TLT/SHY ratio as growth/recession signal.
        Returns: growth_score (-1 to +1, positive = expansionary surprise)"""
